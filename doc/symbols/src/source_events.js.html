 31 //## Update #################################################################################
 32 /**
 33 * @param {number} now
 34 * @param {goog.events.EventTarget} target
 35 * @constructor
 36 * @extends {goog.events.Event}
 37 */
 38 pike.events.Update = function( now, target){
 39 	goog.events.Event.call(this, pike.events.Update.EVENT_TYPE, target);	
 40 	this.now = now;	
 41 };
 42 
 43 goog.inherits( pike.events.Update, goog.events.Event );
 44 
 45 /**
 46  * Event type
 47  * @const
 48  * @type {string}
 49  */
 50 pike.events.Update.EVENT_TYPE = "update";
