 17 //## Entity #####################################
 18 /**
 19 * Create a new Entity
 20 * @param {...function()} components - comma separated functions
 21 * @constructor
 22 * @extends {goog.events.EventTarget}
 23 * @example
 24 * ~~~
 25 * var myEntity = new pike.core.Entity( pike.components.Sprite, pike.components.Mouse );
 26 * ~~~
 27 */
 28 pike.core.Entity = function( components ){
 29 	goog.events.EventTarget.call(this);
 30 	this.id = goog.getUid(this);
 31 		
 32 	/**
 33 	* @type {!goog.events.EventHandler}
 34 	* @protected
 35 	*/
 36 	this.handler = new goog.events.EventHandler(this);
 37 	
 38 	//augment instance
 39 	for(var idx = 0; idx < arguments.length; idx++){
 40 		if (typeof arguments[idx] !== "function"){
 41 	        	throw "Argument is not a function " + arguments[idx];
 42 	    }
 43 		
 44 		arguments[idx].call(this);
 45 	}		
 46 };
 47 
 48 goog.inherits(pike.core.Entity, goog.events.EventTarget);
 49 
 50 /**
 51 * Get boundaries of Entity
 52 * @return {pike.graphics.Rectangle}
 53 */
 54 pike.core.Entity.prototype.getBounds = function(){
 55 	return new pike.graphics.Rectangle(this.x, this.y, this.w, this.h);
 56 };
 57 
 58 /** @inheritDoc */
 59 pike.core.Entity.prototype.disposeInternal = function() {
 60 	pike.core.Entity.superClass_.disposeInternal.call(this);
 61 	this.handler.dispose();
 62 };

 84 /**
 85 * Set size of Viewport
 86 * @param {number} width
 87 * @param {number} height
 88 * @fires {pike.events.ChangeSize} event
 89 */
 90 pike.core.Viewport.prototype.setSize = function( width, height ){
 91 	var oldW = this.viewport_.w;
 92 	var oldH = this.viewport_.h;
 93 	this.viewport_.w = width;
 94 	this.viewport_.h = height;

 98 };

168 //## Timer #####################################
169 /**
170 * Creates a new Timer.
171 * @constructor
172 * @extends {goog.events.EventTarget}
173 */
174 pike.core.Timer = function(){
175 	goog.events.EventTarget.call(this);	
176 	
177 	/**
178 	* @type {!goog.events.EventHandler}
179 	* @protected
180 	*/
181 	this.handler = new goog.events.EventHandler(this);
182 	
183 	this.boundTick_ = goog.bind(this.tick, this);
184 };
185 
186 goog.inherits(pike.core.Timer, goog.events.EventTarget);
187 
188 /**
189  * Starts Timer
190  */
191 pike.core.Timer.prototype.start = function(){
192 	this.stop();	
193 	this.tick();
194 };
195 
196 /**
197  * Stops Timer
198  */
199 pike.core.Timer.prototype.stop = function(){	
200 	if(this.requestID_){    		
201 		window.cancelAnimationFrame( this.requestID_ );		
202 	}	
203 };
204 
205 /**
206  * Tick
207  * @fires {pike.events.Update} event
208  * @fires {pike.events.Render} event
209  */
210 pike.core.Timer.prototype.tick = function(){	
211 	this.dispatchEvent( new pike.events.Update( new Date().getTime(), this));		
212 	this.dispatchEvent( new pike.events.Render( new Date().getTime(), this));	
213 	this.requestID_ = window.requestAnimationFrame( this.boundTick_ );
214 };

217 /**
218 * Create a new Stage 
219 * @constructor
220 * @extends {goog.events.EventTarget}
221 */
222 pike.core.Stage = function(){
223 	goog.events.EventTarget.call(this);
224 	
225 	this.viewport_ = new pike.graphics.Rectangle(0, 0, 0, 0);
226 	this.gameWorld_ = new pike.graphics.Rectangle(0, 0, 0, 0);
227 	
228 	this.layers_ = [];
229 	
230 	this.createRootElement_();	
231 	
232 	/**
233 	* @type {!goog.events.EventHandler}
234 	* @protected
235 	*/
236 	this.handler = new goog.events.EventHandler(this);	
237 };
238 
239 goog.inherits(pike.core.Stage, goog.events.EventTarget);

241 /**
242  * DOM Element id for Viewport
243  * @const
244  * @type {string}
245  */
246 pike.core.Stage.ELEMENT_ID = "pike-stage";

248 /**
249  * Get a layer
250  * @param {string} name - layer name
251  * @return {?pike.layers.Layer}
252  */
253 pike.core.Stage.prototype.getLayer = function( name ){
254 	for(var idx = 0; idx < this.layers_.length; idx++){
255 		if(this.layers_[idx].name == name){
256 			return this.layers_[idx]; 
257 		}		
258 	}
259 	
260 	throw new Error("Layer with name " + name + " does not exist.");
261 };
262

267 pike.core.Stage.prototype.addLayer = function( layer ){
268 	this.setLayerSize_( layer );	
269 	this.layers_.push(layer); //TODO

271 };

315 /**
316  * Create root element
317  * @private
318  */
319 pike.core.Stage.prototype.createRootElement_ = function(){
320 	this.rootElement_ = document.getElementById( pike.core.Stage.ELEMENT_ID );	
321 	if(!this.rootElement_){
322 		this.rootElement_ = document.createElement("div");
323 		this.rootElement_.setAttribute("id", pike.core.Stage.ELEMENT_ID);
324 		document.getElementsByTagName("body")[0].appendChild( this.rootElement_ );		
325 	}else{
326 		this.rootElement_.innerHTML = '';
327 	}
328 	
329 	this.rootElement_.style.position  = "relative";
330 };

332 /**
333  * Sets the layer size
334  * @param {pike.layers.Layer} layer
335  * @private
336  */
337 pike.core.Stage.prototype.setLayerSize_ = function(layer){	
338 	layer.setViewportSize( this.viewport_.w, this.viewport_.h );
339 	layer.setViewportPosition( this.viewport_.x, this.viewport_.y );
340 	layer.setGameWorldSize( this.gameWorld_.w, this.gameWorld_.h );
341 };
342 
343 /**
344  * Set internal viewport size
345  * @param {number} width
346  * @param {number} height
347  * @private
348  */
349 pike.core.Stage.prototype.setViewportSize_ = function(width, height){
350 	if(this.viewport_.w == width
351 	&& this.viewport_.h == height){
352 		return
353 	}
354 	
355 	this.viewport_.w = width;
356 	this.viewport_.h = height;
357 	
358 	this.rootElement_.style.width = this.viewport_.w + "px";
359 	this.rootElement_.style.height = this.viewport_.h + "px";
360 	
361 	if(this.gameWorld_.w < width 
362 	|| this.gameWorld_.h < height ){
363 		this.setGameWorldSize_( width, height);
364 	}
365 	
366 	for(var idx = 0; idx < this.layers_.length; idx++){
367 		this.setLayerSize_(this.layers_[idx]);
368 	}
369 };
370 
371 /**
372  * Set internal gameWorld size
373  * @param {number} width
374  * @param {number} height
375  * @private
376  */
377 pike.core.Stage.prototype.setGameWorldSize_ = function(width, height){
378 	this.gameWorld_.w = Math.max(width, this.viewport_.w);
379 	this.gameWorld_.h = Math.max(height, this.viewport_.h);
380 	
381 	for(var idx = 0; idx < this.layers_.length; idx++){
382 		this.setLayerSize_(this.layers_[idx]);
383 	}
384 };
385 
386 /**
387  * Set internal viewport position
388  * @param {number} x
389  * @param {number} y
390  * @private
391  */
392 pike.core.Stage.prototype.setViewportPosition_ = function(x,y){
393 	this.viewport_.x = x;
394 	this.viewport_.y = y;
395 	
396 	for(var idx = 0; idx < this.layers_.length; idx++){		
397 		this.layers_[idx].setViewportPosition(this.viewport_.x, this.viewport_.y);							
398 	}
399 };
