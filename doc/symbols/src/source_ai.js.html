 10 //## Graph #####################################
 11 /**
 12 * Create a new Graph
 13 * @param {Array<pike.ai.path.Node>} nodes
 14 * @param {Array} connections - matrix of connections
 15 * @constructor
 16 * @example
 17 * ~~~
 18 * function createGraph() {
 19 *	var coords = [[248,76],[205,329],[592,230],[420,410], [95,410],[479,230],[420,16],[555,16]];
 20 *	var matrix =
 21 *		[[1, 1, 0, 0, 0, 0, 0, 0],
 22 *		 [1, 1, 0, 1, 0, 0, 0, 0],
 23 *		 [0, 0, 1, 1, 0, 1, 0, 0],
 24 *		 [0, 1, 1, 1, 1, 1, 1, 0],
 25 *		 [0, 0, 0, 1, 1, 0, 0, 0],
 26 *		 [0, 0, 1, 1, 0, 1, 1, 0],
 27 *		 [0, 0, 0, 1, 0, 1, 1, 1],
 28 *		 [0, 0, 0, 0, 0, 0, 1, 1]];
 29 *		
 30 *	var nodes = [];
 31 *	for (var i = 0; i < coords.length; i++) {
 32 *		nodes.push(new Node(i, coords[i][0], coords[i][1]));
 33 *	}
 34 *	graph = new Graph(nodes, matrix);
 35 * }
 36 * ~~~
 37 */
 38 pike.ai.path.Graph = function(nodes, connections){
 39 	this.nodes_ = nodes;
 40 	
 41 	if (!connections) return;
 42 	
 43 	for (var i = 0; i < connections.length; i++) {
 44         for (var j = 0; j < connections[i].length; j++) {
 45             if (connections[i][j]) {            	
 46                 nodes[i].addConnection(nodes[j], pike.ai.path.Graph.distance(nodes[i], nodes[j]));                              
 47             }
 48         }
 49     }
 50 };

 52 /**
 53  * Find path - A* 
 54  * @param {pike.ai.path.Node} startNode
 55  * @param {pike.ai.path.Node} endNode
 56  */
 57 pike.ai.path.Graph.prototype.findPath = function(startNode, endNode){
 58 		
 59 	/*
 60 	 * Every node has three properties
 61 	 * 1) prev node
 62 	 * 2) estimate cost to finish
 63 	 * 3) route cost
 64 	 */
 65 	function updateNodeValues(node, prevNode, routeCost) {
 66 		node.routeCost = routeCost;
 67 		node.estimatedCost = routeCost + pike.ai.path.Graph.distance(node, endNode);
 68 		node.prevNode = prevNode;			
 69 	}
 70 	
 71 	var openList = [];
 72     var closedList = [];
 73     
 74     startNode.routeCost = 0;
 75     openList.push(startNode);
 76     var routeFound = false;
 77            
 78   while (openList.length > 0) {
 79 	  	        
 80     	var currentNode = openList.sort(function(a, b){return a.estimatedCost - b.estimatedCost; })[0];
 81     	
 82     	if (currentNode == endNode) {    		
 83              routeFound = true;
 84              break;
 85         }
 86     	    	    	 	    	
 87     	currentNode.getConnections().forEach(function( connection ) {
 88     		    		    		      		  
 89     		var node = connection.node;
 90     		var newRouteCost = currentNode.routeCost + connection.weight;
 91     		
 92     		
 93     		    		    	
 94     		if (closedList.indexOf(node) > -1) {
 95                 // The node is in closed list
 96                 if (newRouteCost < node.routeCost) {
 97                 	
 98                     // Remove from closed list
 99                     closedList.splice(closedList.indexOf(node), 1);
100 
101                     updateNodeValues(node, currentNode, newRouteCost);
102                     openList.push(node);
103                 }
104             } else if (openList.indexOf(node) > -1) {
105                 // The node is in open list
106                 if (newRouteCost < node.routeCost) {  
107                 	
108                     updateNodeValues(node, currentNode, newRouteCost);
109                 }
110             } else {
111             	            	            
112                 // The node is not processed
113                 updateNodeValues(node, currentNode, newRouteCost);                                              
114                 openList.push(node);     
115                 
116                 
117             }
118     		    		    	    		    		    	  
119     		if( openList.indexOf(currentNode) != -1 ){
120     			// Remove from open list
121     			openList.splice( openList.indexOf(currentNode), 1);
122     		}
123                        
124             // Add to closed list
125             closedList.push(currentNode);      
126             
127             
128             
129     	});    	    	    	  
130     }
131   
132    var route = [];
133    if (routeFound) {
134        var routeNode = endNode;
135        while (routeNode) {
136            route.push(routeNode);
137            routeNode= routeNode.prevNode;
138        }
139        route.reverse();
140    }
141    
142    // Cleanup, so that old values don't mess around
143    this.nodes_.forEach(function(node) {
144        delete node.routeCost;
145        delete node.estimatedCost;
146        delete node.prevNode;
147    });
148 
149    return route;          	    
150 };

152 /**
153  * @param {pike.ai.path.Node} node1
154  * @param {pike.ai.path.Node} node2
155  * @returns {number}
156  */
157 pike.ai.path.Graph.distance = function(node1, node2) {
158     return Math.sqrt(
159         (node1.x - node2.x)*(node1.x - node2.x) +
160         (node1.y - node2.y)*(node1.y - node2.y));
161 };
162 //## Node #####################################
163 /**
164 * Create a new Node
165 * @param {*} id
166 * @param {number} x - position
167 * @param {number} y - position
168 * @constructor
169 */
170 pike.ai.path.Node = function(id, x, y){
171 	this.id = id;
172 	this.x = x;
173 	this.y = y;
174 	this.connections_ = [];
175 };

177 /**
178  * Add a connection
179  * @param {pike.ai.path.Node} node
180  * @param {number} weight
181  */
182 pike.ai.path.Node.prototype.addConnection = function(node, weight) {
183 	this.connections_.push({ node: node, weight: weight });
184 };
185 
186 /**
187  * Get a connections of the Node
188  * @returns {Array<pike.ai.path.Node>}
189  */
190 pike.ai.path.Node.prototype.getConnections = function() {
191 	return this.connections_;
