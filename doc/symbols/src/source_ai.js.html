 13 //## NumericDecisionNode #####################################
 14 /**
 15  * Create new Decision Node
 16  * Numeric decision is based on the value of the certain agent parameter.
 17  * If value is lesser or equal than the threshold, the leNode is executed otherwise - gNode.
 18  * @param {string} name - agent property name
 19  * @param {number} threshold
 20  * @param {Object} leNode - less then threshold Node
 21  * @param {Object} gNode - grow then threshold Node
 22  * @constructor
 23  * @example
 24  * ~~~
 25  * var hero = new Hero({intelligence:80, strength:30, communication:40});
 26  * 
 27  * function WinAction(){}
 28  *		WinAction.prototype.execute = function( hero ){
 29  *		console.log("Won");
 30  * }
 31  * 
 32  * function LossAction(){}
 33  * 		LossAction.prototype.execute = function( hero ){
 34  * 		console.log("Loss");
 35  * }
 36  * 
 37  * function decisionTree() {
 38  *	var winAction = new WinAction();
 39  *	var lossAction = new LossAction();
 40  *
 41  *	return new NumericDecisionNode("intelligence", 80, 
 42  *		new NumericDecisionNode("strength", 50, lossAction, winAction), 
 43  *		new ProbabilisticDecisionNode(0.5, lossAction, winAction));
 44  * }
 45  * 
 46  * var tree = decisionTree();
 47  * tree.execute( hero );
 48  * ~~~
 49  */
 50 pike.ai.decision.NumericDecisionNode = function (name, threshold, leNode, gNode) {
 51     this.name_ = name;
 52     this.threshold_ = threshold;
 53     this.leNode_ = leNode;
 54     this.gNode_ = gNode;
 55 };
 56 
 57 goog.inherits(pike.ai.decision.NumericDecisionNode, pike.ai.decision.DecisionTreeNode);
 58 
 59 /**
 60  * @inheritDoc
 61  */
 62 pike.ai.decision.NumericDecisionNode.prototype.execute = function(agent, data) {
 63     var node = agent[this.name_] > this.threshold_ ? this.gNode_ : this.leNode_;    
 64     node.execute(agent);
 65 };

 67 //## ProbabilisticDecisionNode #####################################
 68 /**
 69  * Create new Decision Node
 70  * Probabilistic decision is executed depending on the chance value
 71  * 0 - never
 72  * 1 - always
 73  * between 0-1 with the
 74  * @param {number} chance
 75  * @param {Object} trueNode
 76  * @param {Object} falseNode
 77  * @constructor
 78  * @example
 79  * ~~~
 80  * var hero = new Hero({intelligence:80, strength:30, communication:40});
 81  * 
 82  * function WinAction(){}
 83  *		WinAction.prototype.execute = function( hero ){
 84  *		console.log("Won");
 85  * }
 86  * 
 87  * function LossAction(){}
 88  * 		LossAction.prototype.execute = function( hero ){
 89  * 		console.log("Loss");
 90  * }
 91  * 
 92  * function decisionTree() {
 93  *	var winAction = new WinAction();
 94  *	var lossAction = new LossAction();
 95  *
 96  *	return new ProbabilisticDecisionNode(0.8, 
 97  *		new NumericDecisionNode("strength", 50, lossAction, winAction),  
 98  *		new ProbabilisticDecisionNode(0.5, lossAction, winAction));
 99  * }
100  * 
101  * var tree = decisionTree();
102  * tree.execute( hero );
103  * ~~~
104  */
105 pike.ai.decision.ProbabilisticDecisionNode = function(chance, trueNode, falseNode) {
106     this.chance_ = chance;
107     this.trueNode_ = trueNode;
108     this.falseNode_ = falseNode;
109 };
110 
111 goog.inherits(pike.ai.decision.ProbabilisticDecisionNode, pike.ai.decision.DecisionTreeNode);
112 
113 /**
114  * @inheritDoc
115  */
116 pike.ai.decision.ProbabilisticDecisionNode.prototype.execute = function(agent, data) {
117     var node = Math.random() < this.chance_ ? this.trueNode_ : this.falseNode_;
118     node.execute(agent);
119 };

139 //## Graph #####################################
140 /**
141 * Create a new Graph
142 * @param {Array.<pike.ai.path.Node>} nodes
143 * @param {Array} connections - matrix of connections
144 * @constructor
145 * @example
146 * ~~~
147 * function createGraph() {
148 *	var coords = [[248,76],[205,329],[592,230],[420,410], [95,410],[479,230],[420,16],[555,16]];
149 *	var matrix =
150 *		[[1, 1, 0, 0, 0, 0, 0, 0],
151 *		 [1, 1, 0, 1, 0, 0, 0, 0],
152 *		 [0, 0, 1, 1, 0, 1, 0, 0],
153 *		 [0, 1, 1, 1, 1, 1, 1, 0],
154 *		 [0, 0, 0, 1, 1, 0, 0, 0],
155 *		 [0, 0, 1, 1, 0, 1, 1, 0],
156 *		 [0, 0, 0, 1, 0, 1, 1, 1],
157 *		 [0, 0, 0, 0, 0, 0, 1, 1]];
158 *		
159 *	var nodes = [];
160 *	for (var i = 0; i < coords.length; i++) {
161 *		nodes.push(new Node(i, coords[i][0], coords[i][1]));
162 *	}
163 *	graph = new Graph(nodes, matrix);
164 * }
165 * ~~~
166 */
167 pike.ai.path.Graph = function(nodes, connections){
168 	this.nodes_ = nodes;
169 	
170 	if (!connections) return;
171 	
172 	for (var i = 0; i < connections.length; i++) {
173         for (var j = 0; j < connections[i].length; j++) {
174             if (connections[i][j]) {            	
175                 nodes[i].addConnection(nodes[j], pike.ai.path.Graph.distance(nodes[i], nodes[j]));                              
176             }
177         }
178     }
179 };

181 /**
182  * Find path - A* 
183  * @param {pike.ai.path.Node} startNode
184  * @param {pike.ai.path.Node} endNode
185  */
186 pike.ai.path.Graph.prototype.findPath = function(startNode, endNode){
187 		
188 	/*
189 	 * Every node has three properties
190 	 * 1) prev node
191 	 * 2) estimate cost to finish
192 	 * 3) route cost
193 	 */
194 	function updateNodeValues(node, prevNode, routeCost) {
195 		node.routeCost = routeCost;
196 		node.estimatedCost = routeCost + pike.ai.path.Graph.distance(node, endNode);
197 		node.prevNode = prevNode;			
198 	}
199 	
200 	var openList = [];
201     var closedList = [];
202     
203     startNode.routeCost = 0;
204     openList.push(startNode);
205     var routeFound = false;
206            
207   while (openList.length > 0) {
208 	  	        
209     	var currentNode = openList.sort(function(a, b){return a.estimatedCost - b.estimatedCost; })[0];
210     	
211     	if (currentNode == endNode) {    		
212              routeFound = true;
213              break;
214         }
215     	    	    	 	    	
216     	currentNode.getConnections().forEach(function( connection ) {
217     		    		    		      		  
218     		var node = connection.node;
219     		var newRouteCost = currentNode.routeCost + connection.weight;
220     		
221     		
222     		    		    	
223     		if (closedList.indexOf(node) > -1) {
224                 // The node is in closed list
225                 if (newRouteCost < node.routeCost) {
226                 	
227                     // Remove from closed list
228                     closedList.splice(closedList.indexOf(node), 1);
229 
230                     updateNodeValues(node, currentNode, newRouteCost);
231                     openList.push(node);
232                 }
233             } else if (openList.indexOf(node) > -1) {
234                 // The node is in open list
235                 if (newRouteCost < node.routeCost) {  
236                 	
237                     updateNodeValues(node, currentNode, newRouteCost);
238                 }
239             } else {
240             	            	            
241                 // The node is not processed
242                 updateNodeValues(node, currentNode, newRouteCost);                                              
243                 openList.push(node);     
244                 
245                 
246             }
247     		    		    	    		    		    	  
248     		if( openList.indexOf(currentNode) != -1 ){
249     			// Remove from open list
250     			openList.splice( openList.indexOf(currentNode), 1);
251     		}
252                        
253             // Add to closed list
254             closedList.push(currentNode);      
255             
256             
257             
258     	});    	    	    	  
259     }
260   
261    var route = [];
262    if (routeFound) {
263        var routeNode = endNode;
264        while (routeNode) {
265            route.push(routeNode);
266            routeNode= routeNode.prevNode;
267        }
268        route.reverse();
269    }
270    
271    // Cleanup, so that old values don't mess around
272    this.nodes_.forEach(function(node) {
273        delete node.routeCost;
274        delete node.estimatedCost;
275        delete node.prevNode;
276    });
277 
278    return route;          	    
279 };

281 /**
282  * Serialize the graph
283  * @return {Object} json
284  */
285 pike.ai.path.Graph.serialize = function() {
286     var data = {};
287     data.nodes = [];
288     this.nodes_.forEach(function(it) {
289         data.nodes.push([it.x, it.y]);
290     });
291     return JSON.stringify(data);
292 };

294 /**
295  * @param {pike.ai.path.Node} node1
296  * @param {pike.ai.path.Node} node2
297  * @returns {number}
298  */
299 pike.ai.path.Graph.distance = function(node1, node2) {
300     return Math.sqrt(
301         (node1.x - node2.x)*(node1.x - node2.x) +
302         (node1.y - node2.y)*(node1.y - node2.y));
303 };
