  9 //## Rectangle #####################################
 10 /**
 11  * Create a new Rectangle
 12  * @param {number} x
 13  * @param {number} y
 14  * @param {number} width
 15  * @param {number} height
 16  * @constructor
 17  */
 18 pike.graphics.Rectangle = function( x, y, width, height){
 19 	this.x = x || 0;
 20     this.y = y || 0;
 21     this.w = width || 0;
 22     this.h = height || 0;	
 23 };
 24 
 25 /**
 26  * Copy rectangle
 27  * @return {pike.graphics.Rectangle}
 28  */
 29 pike.graphics.Rectangle.prototype.copy = function(){
 30 	return new pike.graphics.Rectangle( this.x, this.y, this.w, this.h );
 31 };

 33 /**
 34  * Equals rectangles
 35  * @param {pike.graphics.Rectangle} r2 - rectangle
 36  * @return {boolean}
 37  */
 38 pike.graphics.Rectangle.prototype.equals = function( r2 ){
 39 	return (this.x == r2.x
 40 	     && this.y == r2.y
 41 	     && this.w == r2.w
 42 	     && this.h == r2.h);	
 43 };

 45 /**
 46  * Converts Rectangle to string
 47  * @return {string}
 48  */
 49 pike.graphics.Rectangle.prototype.toString = function(){
 50 	return ("[x=" + this.x + ", y=" + this.y + ", width=" + this.w + ", height=" + this.h + "]");
 51 };

 53 /**
 54  * Determines whether to intersect
 55  * @param {pike.graphics.Rectangle} r2 - rectangle
 56  * @return {boolean}
 57  */
 58 pike.graphics.Rectangle.prototype.intersects = function( r2 ){
 59 	return (this.x <= r2.x + r2.w 
 60 			&& this.x + this.w >= r2.x 
 61 			&& this.y <= r2.y + r2.h 
 62 			&& this.y + this.h >= r2.y);
 63 };
 64 
 65 /**
 66  * The intersection of two rectangles
 67  * @param {pike.graphics.Rectangle} r2
 68  * @return {?pike.graphics.Rectangle}
 69  */
 70 pike.graphics.Rectangle.prototype.intersection = function( r2 ){
 71 	 var x = Math.max(this.x, r2.x);
 72 	 var y = Math.max(this.y, r2.y);
 73 	 var width = Math.min(this.x + this.w, r2.x + r2.w) - x;
 74 	 var height = Math.min(this.y + this.h, r2.y + r2.h) - y;
 75 	
 76 	  if (width <= 0 || height <= 0)
 77 		  return null;
 78 	
 79 	  return new pike.graphics.Rectangle(x, y, width, height);
 80 };

 82 /**
 83  * Determines that the rectangle(this) fully covers the given rectangle(r2)
 84  * @param {pike.graphics.Rectangle} r2
 85  * @return {boolean}
 86  */
 87 pike.graphics.Rectangle.prototype.covers = function( r2 ){
 88 	return (r2.x >= this.x 
 89 			&& r2.y >= this.y 
 90 			&& r2.x + r2.w <= this.x + this.w 
 91 			&& r2.y + r2.h <= this.y + this.h);	
 92 };
 93 
 94 /**
 95  * Determines that rectangle contains point
 96  * @param {number} x
 97  * @param {number} y
 98  * @return {boolean}
 99  */
100 pike.graphics.Rectangle.prototype.containsPoint = function(x, y){
101 	return x >= this.x 
102 		&& x < this.x + this.w 
103 		&& y >= this.y 
104 		&& y < this.y + this.h;
105 };

107 /**
108  * Creates rectangular container, which covers both rectangles.
109  * Assumes that neither of rectangles is zero-area
110  * @param {pike.graphics.Rectangle} r2
111  * @return {pike.graphics.Rectangle}
112  */
113 pike.graphics.Rectangle.prototype.convexHull = function( r2 ){
114 	var x = Math.min(this.x, r2.x);
115     var y = Math.min(this.y, r2.y);
116     var width = Math.max(this.x + this.w, r2.x + r2.w) - x;
117     var height = Math.max(this.y + this.h, r2.y + r2.h) - y;
118 
119     return new pike.graphics.Rectangle(x, y, width, height);
120 };

122 /**
123  * Test which cells are overlapped by the rectangle
124  * @param {number} cellW
125  * @param {number} cellH
126  * @param {number} cellsInRow
127  * @param {number} cellsInColumn
128  * @return {pike.graphics.Rectangle}
129  */
130 pike.graphics.Rectangle.prototype.getOverlappingGridCells = function(cellW, cellH, cellsInRow, cellsInColumn){
131 	var rectX = Math.max(0, Math.floor(this.x/cellW));
132     var rectY = Math.max(0, Math.floor(this.y/cellH));
133     var rectWidth = Math.min(cellsInRow - rectX, Math.floor((this.x + this.w)/cellW) - rectX + 1);
134     var rectHeight = Math.min(cellsInColumn - rectY, Math.floor((this.y + this.h)/cellH) - rectY + 1);
135     return new pike.graphics.Rectangle(rectX, rectY, rectWidth, rectHeight);
136 };

139 /**
140 * Cluster 
141 * @param {number} clusterSize - side of the square in px
142 * @param {number} width - total width of the clusters
143 * @param {number} height - total height of the clusters
144 * @constructor
145 */
146 pike.graphics.Cluster = function( clusterSize, width, height ){
147 	
148 	this.clusterSize_ = clusterSize;	
149 	this.bounds_ = new pike.graphics.Rectangle(0, 0, width, height);
150 	
151 	/* the clusters arranged into the grid */
152 	this.clusters_= [];
153 
154     /* object id to Rect - the cluster bounds */
155     this.idToClusterBounds_ = {};        
156 };
157 
158 /**
159  * Set size
160  * @param {number} width - total width of the clusters
161  * @param {number} height - total height of the clusters
162  */
163 pike.graphics.Cluster.prototype.setSize = function(width, height){
164 	this.bounds_.w = width;
165 	this.bounds_.h = height;
166 };

198 /**
199  * Build a clusters
200  */
201 pike.graphics.Cluster.prototype.build = function(){
202 	this.clusters_ = [];	
203 	for (var i = 0; i < Math.ceil(this.bounds_.h/this.clusterSize_); i++) {		
204         this.clusters_[i] = [];        
205         for (var j = 0; j < Math.ceil(this.bounds_.w/this.clusterSize_); j++) {
206             this.clusters_[i][j] = [];            
207         }
208     }	
209 };
210 
211 /**
212  * Adds an entity to the clusters
213  * @param {pike.core.Entity} entity
214  	* @param {?pike.graphics.Rectangle} clusterBounds
215  * @returns {pike.graphics.Rectangle}
216  */
217 pike.graphics.Cluster.prototype.addToClusters = function( entity, clusterBounds ){
218 	clusterBounds = clusterBounds || entity.getBounds().getOverlappingGridCells(
219         this.clusterSize_, 
220         this.clusterSize_, 
221         this.clusters_[0].length, 
222         this.clusters_.length);
223 		 
224     for (var clusterY = clusterBounds.y; clusterY < clusterBounds.y + clusterBounds.h; clusterY++) {
225         for (var clusterX = clusterBounds.x; clusterX < clusterBounds.x + clusterBounds.w; clusterX++) {
226             this.clusters_[clusterY][clusterX].push( entity );              
227         }
228     }
229     //save bounds in cluster
230     this.idToClusterBounds_[entity.id] = clusterBounds;
231     if(goog.DEBUG) window.console.log("[pike.graphics.Cluster] entity " +  entity.id +" is added to cluster " + clusterBounds);
232     return clusterBounds;	
233 };
234 
235 /**
236  * Removes an entity from the clusters
237  * @param {pike.core.Entity} entity
238  * @param {?pike.graphics.Rectangle} clusterBounds
239  */
240 pike.graphics.Cluster.prototype.removeFromClusters = function( entity, clusterBounds ){
241 	clusterBounds = clusterBounds || this.idToClusterBounds_[entity.id];
242     for (var clusterY = clusterBounds.y; clusterY < clusterBounds.y + clusterBounds.h; clusterY++) {
243         for (var clusterX = clusterBounds.x; clusterX < clusterBounds.x + clusterBounds.w; clusterX++) {            
244             goog.array.remove(this.clusters_[clusterY][clusterX], entity);            
245         }
246     }
247     
248     if(goog.DEBUG) window.console.log("[pike.graphics.Cluster] entity " + entity.id + " is removed from cluster " + clusterBounds);
249 };
