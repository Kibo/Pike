 21 /**
 22  * Create a new layer
 23  * @param {string} name
 24  * @constructor
 25  * @extends {goog.events.EventTarget}
 26  */
 27 pike.layers.Layer = function( name ){
 28 	goog.events.EventTarget.call(this);
 29 	
 30 	this.name = name;
 31 	
 32 	/**
 33 	* @type {!goog.events.EventHandler}
 34 	* @protected
 35 	*/
 36 	this.handler = new goog.events.EventHandler(this);
 37 	
 38 	this.viewport_ = new pike.graphics.Rectangle(0,0,0,0);
 39 	this.gameWorld_ = new pike.graphics.Rectangle(0,0,0,0);
 40 	
 41 	this.entities_ = [];
 42 	this.screen_ = {};
 43 	this.offScreen_ = {}; 
 44 	
 45 	this.screen_.canvas = document.createElement("canvas");	
 46 	this.screen_.canvas.style.position  = "absolute";
 47 	this.screen_.canvas.style.top  = "0";
 48 	this.screen_.canvas.style.left  = "0";        		
 49 	this.screen_.context = this.screen_.canvas.getContext('2d');
 50 	this.screen_.isDirty = false;
 51 	        		        	
 52 	this.offScreen_.canvas = document.createElement("canvas");  	
 53 	this.offScreen_.context = this.offScreen_.canvas.getContext('2d');
 54 	this.offScreen_.isDirty = false; 	
 55 };

 60 /**
 61  * On render handler
 62  */
 63 pike.layers.Layer.prototype.onRender = function(){
 64 	
 65 	if(this.hasDirtyManager()){
 66 		this.renderDirty_();
 67 		
 68 	}else if( this.offScreen_.isDirty ){
 69 		this.renderOffScreen_();
 70 	}
 71 	
 72 	if(this.screen_.isDirty){
 73 		this.renderScreen_();
 74 	}	
 75 };

 77 /**
 78  * Renders dirty area
 79  * @private
 80  */
 81 pike.layers.Layer.prototype.renderDirty_ = function(){
 82 	if( this.dirtyManager.isClean() ){
 83 		return;
 84 	}
 85 	
 86 	this.offScreen_.context.clearRect(
 87 			this.dirtyManager.getDirtyRectangle().x,
 88 			this.dirtyManager.getDirtyRectangle().y,
 89 			this.dirtyManager.getDirtyRectangle().w,
 90 			this.dirtyManager.getDirtyRectangle().h
 91 	);
 92 	
 93 	this.dispatchEvent( new pike.events.Render( new Date().getTime(), this));
 94 									
 95 	if(!this.screen_.isDirty){
 96 		this.screen_.context.clearRect(
 97 			(this.dirtyManager.getDirtyRectangle().x - this.viewport_.x),
 98 			(this.dirtyManager.getDirtyRectangle().y - this.viewport_.y),
 99 			this.dirtyManager.getDirtyRectangle().w,
100 			this.dirtyManager.getDirtyRectangle().h						
101 		);
102 		
103 		this.screen_.context.drawImage(
104 				this.offScreen_.canvas,
105 				this.dirtyManager.getDirtyRectangle().x,
106 				this.dirtyManager.getDirtyRectangle().y,
107 				this.dirtyManager.getDirtyRectangle().w,
108 				this.dirtyManager.getDirtyRectangle().h,
109 				
110 				(this.dirtyManager.getDirtyRectangle().x - this.viewport_.x),
111 				(this.dirtyManager.getDirtyRectangle().y - this.viewport_.y),
112 				this.dirtyManager.getDirtyRectangle().w,
113 				this.dirtyManager.getDirtyRectangle().h						
114 		);										
115 	}
116 						
117 	this.dirtyManager.clear();		
118 };

120 /**
121  * Renders offscreen
122  * @private
123  */
124 pike.layers.Layer.prototype.renderOffScreen_ = function(){
125 	this.offScreen_.context.clearRect( 0, 0, this.gameWorld_.w, this.gameWorld_.h );
126 	
127 	this.dispatchEvent( new pike.events.Render( new Date().getTime(), this) );
128 			
129 	this.offScreen_.isDirty = false;
130 	this.screen_.isDirty = true;
131 	if(goog.DEBUG) window.console.log("[pike.core.Layer] " + this.name + " redraw offScreen");
132 };

145 	this.screen_.isDirty = false;
146 	if(goog.DEBUG) window.console.log("[pike.core.Layer] " + this.name + " redraw screen");

267 /**
268  * Create a new Cluster layer
269  * @param {string} name
270  * @param {numner} clusterSize - px
271  * @constructor
272  * @extends {pike.layers.Layer}
273  */
274 pike.layers.ClusterLayer = function( name, clusterSize ){
275 	pike.layers.Layer.call(this, name);
276 		   
277     this.clusters_ = new pike.graphics.Cluster(clusterSize, 0, 0);
278 
279     /* currently visible clusters */
280     this.visibleClusterBounds_ = {};
281 
282     /* the sorted array of objects from active clusters, no duplicates*/
283     this.cache_ = [];
284 
285     /* true if cache needs to be fully rebuilt */
286     this.cacheDirty_ = true;
287 
288     /* true if cache only needs to be sorted (when object moved for example) */
289     this.cacheUnsorted_ = false;   
290             
291 };

296 /**
297  * Cluster instance attached to this layer
298  * @return {pike.graphics.Cluster} 
299  */
300 pike.layers.ClusterLayer.prototype.getCluster = function(){
301 	return this.clusters_;
302 };

304 /**
305  * @override
306  */
307 pike.layers.ClusterLayer.prototype.dispatchEvent = function( e ){
308 	
309 	if (this.cacheDirty_) {			
310 		this.resetCache_();
311 	} else if (this.cacheUnsorted_) {		
312 		this.sortCache_();
313 	}
314 	
315 	for (var i = 0; i < this.cache_.length; i++) {
316 		var entity = this.cache_[i];
317 		if (entity.getBounds().intersects(this.viewport_)) {
318 			entity.dispatchEvent(e);
319 		}
320 	}
321 
322 	goog.events.EventTarget.prototype.dispatchEvent.call(this, e);
323 };

325 /**
326  * Add a entity to the layer
327  * @param {pike.core.Entity} entity
328  * @fires {pike.events.NewEntity} newentity
329  * @override
330  */
331 pike.layers.ClusterLayer.prototype.addEntity = function( entity ){	
332 	pike.layers.Layer.prototype.addEntity.call(this, entity);	
333 	this.handler.listen(entity, pike.events.ChangePosition.EVENT_TYPE, goog.bind( this.onEntityMove, this ));
334 		
335 	//Cluster will build after attached to stage
336 	if( this.clusters_.getClusters().length == 0 ){
337 		return
338 	}
339 		
340 	var clusters = this.clusters_.addToClusters( entity );
341 	if (clusters.intersects(this.visibleClusterBounds_)) {			
342 		this.cache_.push( entity );
343 		this.cacheUnsorted_ = true;			
344 	}
345 };
346 
347 /**
348  * Remove entity
349  * @param {pike.core.Entity} entity
350  * @fires {pike.events.RemoveEntity} removeentity
351  * @override
352  */
353 pike.layers.ClusterLayer.prototype.removeEntity = function( entity ){
354 	this.clusters_.removeFromClusters(entity);	
355 	pike.layers.Layer.prototype.removeEntity.call(this, entity);
356 };

358 pike.layers.ClusterLayer.prototype.resetCache_ = function(){
359 	var cache = this.cache_ = [];
360     for (var i = this.visibleClusterBounds_.y; i < this.visibleClusterBounds_.y + this.visibleClusterBounds_.h; i++) {
361         for (var j = this.visibleClusterBounds_.x; j < this.visibleClusterBounds_.x + this.visibleClusterBounds_.w; j++) {
362             var cluster = this.clusters_.getClusters()[i][j];
363             for (var k = 0; k < cluster.length; k++) {
364                 if (cache.indexOf(cluster[k]) == -1) {                	
365                     cache.push(cluster[k]);
366                 }
367             }
368         }
369     }
370     
371     if(goog.DEBUG) window.console.log("[pike.layers.ClusterLayer] resetcache");
372     
373     this.sortCache_();
374     this.cacheDirty_ = false;
375     this.cacheUnsorted_ = false;    
376 };
377 
378 /**
379  * @private
380  */
381 pike.layers.ClusterLayer.prototype.sortCache_ = function() {
382     this.cache_.sort(function(a, b) {
383         var aBounds = a.getBounds();
384         var bBounds = b.getBounds();
385 
386         return (aBounds.y + aBounds.h) - (bBounds.y + bBounds.h);
387     });
388         
389     this.cacheUnsorted_ = false;
390     if(goog.DEBUG) window.console.log("[pike.layers.ClusterLayer] sortcache");
391 };

393 /**
394  * Recreate the clusters
395  * @private
396  */
397 pike.layers.ClusterLayer.prototype.resetClusters_ = function(){		
398 	this.clusters_.build();
399 		
400 	if(goog.DEBUG) window.console.log("[pike.layers.ClusterLayer] resetcluster");
401 	
402 	for (var i=0; i < this.entities_.length; i++) {
403         var entity = this.entities_[i];  
404         this.clusters_.addToClusters(entity); 
405     }		
406 };

408 pike.layers.ClusterLayer.prototype.updateVisibleClusters_ = function(){
409 	var newRect = this.viewport_.getOverlappingGridCells(
410 			this.clusters_.getClusterSize(),
411 			this.clusters_.getClusterSize(),
412 			this.clusters_.getClusters()[0].length,
413 			this.clusters_.getClusters().length);
414 
415     if (!newRect.equals(this.visibleClusterBounds_)) {
416         this.visibleClusterBounds_ = newRect;
417         this.cacheDirty_ = true;        
418     }	
419 };

460 /**
461  * on entity move handler
462  * @param {pike.events.Move} e
463  */
464 pike.layers.ClusterLayer.prototype.onEntityMove = function(e){	
465 	var entity = e.target;
466 	var newClusters = entity.getBounds().getOverlappingGridCells(
467 			this.clusters_.getClusterSize(),
468 			this.clusters_.getClusterSize(),
469 			this.clusters_.getClusters()[0].length,
470 			this.clusters_.getClusters().length);
471 	
472 	var oldClusters = this.clusters_.getIdToClusterBounds(entity.id);
473 	
474 	if (!oldClusters.equals(newClusters)) {
475 		this.moveObjectBetweenClusters_(entity, oldClusters, newClusters);		
476 	}
477 		
478 	if (newClusters.intersects(this.visibleClusterBounds_) && e.y != e.oldY) {
479 		this.cacheUnsorted_ = true;		
480 	}	
481 };
482 
483 pike.layers.ClusterLayer.prototype.moveObjectBetweenClusters_ = function( entity, oldClusters, newClusters) {
484 	this.clusters_.removeFromClusters(entity, oldClusters);
485 	this.clusters_.addToClusters(entity, newClusters);
486 	this.clusters_.setIdToClusterBounds(entity.id, newClusters);
487 	
488 	// If object has left the screen, remove from cache
489 	if (newClusters.intersects(this.visibleClusterBounds_)) {
490 		
491 		if(!goog.array.contains(this.cache_, entity)){
492 			this.cache_.push(entity);
493 		}
494 		
495 	} else {		
496 		goog.array.remove(this.cache_, entity);
497 	}
498 };

558 //## DirtyManager #################################
559 /**
560  * Create a new DirtyManager
561  * @param {number} allDirtyThreshold - the value is the number between 0 and 1.
562  * @constructor
563  */
564 pike.layers.DirtyManager = function( allDirtyThreshold ){
565 		
566 	this.viewport_ = new pike.graphics.Rectangle(0, 0, 0, 0);
567 	
568 	/**
569 	* @type {!goog.events.EventHandler}
570 	* @protected
571 	*/
572 	this.handler = new goog.events.EventHandler(this);
573 	
574 	// Current dirty rectangle that covers all smaller dirty areas
575 	this.dirtyRect_ = null;
576 	
577 	this.allDirtyThreshold_ = allDirtyThreshold == undefined ? .5 : allDirtyThreshold;
578 	
579 	// true when we have reached the threshold
580 	this.allDirty_ = true;
581 	
582 	this.markAllDirty();
583 };
584 
585 /**
586  * Returns the current dirty rectangle that covers all registered dirty areas
587  * @return {pike.graphics.Rectangle}
588  */
589 pike.layers.DirtyManager.prototype.getDirtyRectangle = function(){
590 	return this.dirtyRect_;
591 };
592 
593 /**
594  * Mark all viewport as dirty
595  */
596 pike.layers.DirtyManager.prototype.markAllDirty = function(){
597 	this.allDirty_ = true;
598 	this.dirtyRect_ = this.viewport_.copy();
599 };
600 
601 /**
602  * Mark the given area as dirty.
603  * @param {?pike.graphics.Rectangle} rect
604  */
605 pike.layers.DirtyManager.prototype.markDirty = function( rect ){	
606 	if ( !(rect.w || rect.h) || this.allDirty_) {
607 		return;
608 	}
609 		
610 	// We are only interested in the rectangles that intersect the viewport
611 	rect = this.viewport_.intersection( rect );	
612 	
613 	if (!rect) {
614 		return;
615 	}
616 															
617 	if (this.dirtyRect_) {
618 		this.dirtyRect_ = this.dirtyRect_.convexHull(rect);		
619 	} else {					
620 		this.dirtyRect_ = rect;		
621 	}
622 		
623 	// Check for threshold. If it is reached, mark the whole screen dirty
624 	if (this.dirtyRect_.w * this.dirtyRect_.h > this.allDirtyThreshold_ * this.viewport_.w * this.viewport_.h){	
625 			this.markAllDirty();
626 	}	
627 };
628

633 pike.layers.DirtyManager.prototype.isClean = function(){
634 	return !(this.dirtyRect_);
635 };
