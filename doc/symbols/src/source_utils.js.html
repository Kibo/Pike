  6 goog.provide('pike.Utils');
  7 
  8 goog.require('pike.input.InputHandlerBase');
  9 goog.require('pike.input.MouseInputHandler');
 10 goog.require('pike.input.TouchInputHandler');
 11 
 12 /**
 13  * Is touch device
 14  * @return {boolean}
 15  */
 16 pike.Utils.isTouchDevice = function(){
 17 	return ('ontouchstart' in document.documentElement);	
 18 };
 19 
 20 /**
 21  * Get input handler
 22  * mouse or touch
 23  * @return {pike.input.MouseInputHandler | pike.input.TouchInputHandler} 
 24  */
 25 pike.Utils.getDeviceInputHandler = function(){
 26 	return pike.Utils.isTouchDevice() ? new pike.input.TouchInputHandler() : new pike.input.MouseInputHandler();
 27 };
