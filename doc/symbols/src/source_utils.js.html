  6 goog.provide('pike.Utils');

  8 goog.require('goog.style');

 10 /**
 11  * Is touch device
 12  * @return {boolean}
 13  */
 14 pike.Utils.isTouchDevice = function(){
 15 	return ('ontouchstart' in document.documentElement);	
 16 };
 17 
 18 /**
 19  * Calculate a input coordinates
 20  * @param {Object} e - DOM event
 21  * @param {Object} element - DOM element
 22  * @returns {Object}
 23  */
 24 pike.Utils.getInputCoordinates = function(e, element){
 25 	var coords = e.targetTouches ? e.targetTouches[0] : e;
 26 	
 27 	var offset = goog.style.getPageOffset(element);	
 28 	
 29 	return {
 30 		posX: (coords.pageX || coords.clientX + document.body.scrollLeft ) - offset.x,
 31 		posY: (coords.pageY || coords.clientY + document.body.scrollTop ) - offset.y
 32 	};
 33 };
